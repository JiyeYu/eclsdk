[docs]    def update(self, session, firewall_id, **attrs):
        """Update a firewall."""

        uri = utils.urljoin(self.base_path, firewall_id)
        resp = session.put(
            uri, endpoint_filter=self.service,
            json=attrs
        )
        self._translate_response(resp, has_body=True)
        return self
